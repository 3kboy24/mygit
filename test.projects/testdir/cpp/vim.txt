function! CheckChineseMark()  
    "依次检查  
    if search('。')  
        let s:line=search('。')  
        execute s:line . "s/。/\./g" 
    endif  
 
    if search('，')  
        let s:line=search('，')  
        execute s:line . "s/，/,/g" 
    endif  
 
    if search('；')  
        let s:line=search('；')  
        execute s:line . "s/；/,/g" 
 
    endif  
      
    if  search('？')  
        let s:line=search('？')  
        execute s:line . "s/？/?/g" 
    endif  
 
    if search('：')  
        let s:line=search('：')  
        execute s:line . "s/：/\:/g" 
    endif  
 
    if search('‘')  
        let s:line=search('‘')  
        execute s:line . "s/‘/\'/g" 
    endif  
 
    if search('’')  
        let s:line=search('’')  
        execute s:line . "s/’/\'/g" 
    endif  
 
    if search('”')  
        let s:line=search('”')  
        execute s:line . "s/”/\"/g" 
    endif  
 
    if search('“')  
        let s:line=search('“')  
        execute s:line . "s/“/\"/g" 
    endif  
 
    if search('《')  
        let s:line=search('《')  
        execute s:line . "s/《/\</g" 
    endif  
 
    if search('》')  
        let s:linie=search('》')  
        execute s:line . "s/》/\>/g" 
    endif  
 
    if search('——')  
        let s:line=search('——')  
        execute s:line . "s/——/-/g" 
    endif  
 
    if search('）')  
        let s:line=search('）')  
        execute s:line . "s/）/\)/g" 
    endif  
 
    if search('（')  
        let s:line=search('（')  
        execute s:line . "s/（/\(/g" 
    endif  
 
    if search('……')  
        let s:line=search('……')  
        execute s:line . "s/……/^/g" 
    endif  
 
    if search('￥')  
        let s:line=search('￥')  
        execute s:line . "s/￥/$/g" 
    endif  
 
    if search('！')  
        let s:line=search('！')  
        execute s:line . "s/！/!/g" 
    endif  
 
    if  search('·')  
        let s:line=search('·')  
        execute s:line . "s/·/`/g" 
    endif  
 
endfunction  
unmap <C-S>  
map! <C-S> <ESC>:call CheckChineseMark()<ESC>:w<ESC>a
